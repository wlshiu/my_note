USER_SECTIONS .section0, .section1, .section2
ROM 0x0
{
RAM 0x0
{
*(+RO, +RW, +ZI)
STACK = 0xA00000
}
}

ROM_OVLY 0x14000
{
RAM2 0x4000 OVERLAY 0x0
{
.overlay0 {* (.section0)}
.overlay1 {* (.section1)}
.overlay2 {* (.section2)}
}
}
