easylzma
by lloyd
An easy to use, tiny, public domain, C wrapper library around Igor Pavlov's work that can be used to compress and extract lzma files.

Overview

easylzma is a small and portable C library which wraps Igor Pavlov's LZMA reference implementation. It comes with command line tools to compress and uncompress both lzip and LZMA-Alone disk formats. It's also as fast as any implementation out there (largely because it's based on Igor's recent C implementation of LZMA).

  1. /* 
  2.  * an example of basic LZMA compression to and from memory buffers 
  3.  * using the easylzma library. 
  4.  */  
  5.    
  6. #include "easylzma/compress.h"  
  7.    
  8. #include <string.h>  
  9. #include <assert.h>  
  10.    
  11. struct dataStream  
  12. {  
  13.     const unsigned char * inData;  
  14.     size_t inLen;  
  15.    
  16.     unsigned char * outData;  
  17.     size_t outLen;  
  18. };  
  19.   
  20. /* an input callback that will be passed to elzma_compress_run(), 
  21.  * it reads from a memory buffer */  
  22. static int  
  23. inputCallback(void *ctx, void *buf, size_t * size)  
  24. {  
  25.     size_t rd = 0;  
  26.     struct dataStream * ds = (struct dataStream *) ctx;  
  27.     assert(ds != NULL);  
  28.       
  29.     rd = (ds->inLen < *size) ? ds->inLen : *size;  
  30.    
  31.     if (rd > 0) {  
  32.         memcpy(buf, (void *) ds->inData, rd);  
  33.         ds->inData += rd;  
  34.         ds->inLen -= rd;  
  35.     }  
  36.    
  37.     *size = rd;  
  38.    
  39.     return 0;  
  40. }  
  41.   
  42. /* an ouput callback that will be passed to elzma_compress_run(), 
  43.  * it reallocs and writes to a memory buffer */  
  44. static size_t  
  45. outputCallback(void *ctx, const void *buf, size_t size)  
  46. {  
  47.     struct dataStream * ds = (struct dataStream *) ctx;  
  48.     assert(ds != NULL);  
  49.       
  50.     if (size > 0) {  
  51.         ds->outData = realloc(ds->outData, ds->outLen + size);  
  52.         memcpy((void *) (ds->outData + ds->outLen), buf, size);  
  53.         ds->outLen += size;  
  54.     }  
  55.    
  56.     return size;  
  57. }  
  58.   
  59. /* a function that will compress data using a 1mb dictionary and a 
  60.  * client specified encoding format (one of ELZMA_lzip or ELZMA_lzma) */  
  61. int  
  62. simpleCompress(elzma_file_format format, const unsigned char * inData,  
  63.                size_t inLen, unsigned char ** outData,  
  64.                size_t * outLen)  
  65. {  
  66.     int rc;  
  67.     elzma_compress_handle hand;  
  68.    
  69.     /* allocate compression handle */  
  70.     hand = elzma_compress_alloc();  
  71.     assert(hand != NULL);  
  72.   
  73.     /* configure the compression run with mostly default parameters  */   
  74.     rc = elzma_compress_config(hand, ELZMA_LC_DEFAULT,  
  75.                                ELZMA_LP_DEFAULT, ELZMA_PB_DEFAULT,  
  76.                                5, (1 << 20) /* 1mb */,  
  77.                                format, inLen);  
  78.   
  79.     /* fail if we couldn't allocate */    
  80.     if (rc != ELZMA_E_OK) {  
  81.         elzma_compress_free(&hand);  
  82.         return rc;  
  83.     }  
  84.    
  85.     /* now run the compression */  
  86.     {  
  87.         /* set up the context structure that will be passed to 
  88.          * stream callbacks */    
  89.         struct dataStream ds;  
  90.         ds.inData = inData;  
  91.         ds.inLen = inLen;  
  92.         ds.outData = NULL;  
  93.         ds.outLen = 0;  
  94.   
  95.         /* run the streaming compression */   
  96.         rc = elzma_compress_run(hand, inputCallback, (void *) &ds,  
  97.                                 outputCallback, (void *) &ds);  
  98.           
  99.         if (rc != ELZMA_E_OK) {  
  100.             if (ds.outData != NULL) free(ds.outData);  
  101.             elzma_compress_free(&hand);  
  102.             return rc;  
  103.         }  
  104.    
  105.         *outData = ds.outData;  
  106.         *outLen = ds.outLen;  
  107.     }  
  108.    
  109.     return rc;  
  110. }
