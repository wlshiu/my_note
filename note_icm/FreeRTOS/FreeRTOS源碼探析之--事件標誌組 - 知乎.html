事件是一種實現任務間通信的機制，主要用於實現多任務間的同步，但事件通信只能是事件類型的通信，無數據傳輸。

1 基礎概述

1.1 基本作用

事件標誌的作用類似於全局型的flag，多個標誌組合在一起構成事件標誌組，這裡先分析一下事件標誌組於全局flag的區別：

  • 使用事件標誌組可以讓 FreeRTOS內核有效地管理任務，而全局變量是無法做到的，任務的超時等機制需要用戶自己去實現
  • 使用了全局變量就要防止多任務的訪問衝突，而使用事件標誌組則處理好了這個問題，用戶無需擔心
  • 使用事件標誌組可以有效地解決中斷服務程序和任務之間的同步問題

1.2 運行原理



如上圖：

  • 任務1在一個循環中等待事件的到來，等待時處於阻塞狀態，即任務掛起狀態。阻塞的時間可用設置。當有事件位被置位時，執行處理事件。
  • 任務1的處理事件可以觸發事件（給自身的任務觸發事件），但這種方式觸發後，只能等程序再次運行到xEventGroupWaitBits才能處理
  • 任務2或其它任務可以觸發事件（這是比較常用的用法），這樣，在任務2觸發事件後，任務1可以從阻塞態變為就緒態，如果任務1的優先級較高，則任務1開始運行，執行處理事件，實現任務間的同步
  • 中斷中也可以可以觸發事件（這也是比較常用的用法），作用也是實現同步，只是需要使用加FromISR後綴的API函數

1.3 事件組的最大位數

事件標誌存儲在一個EventBits_t類型的變量中，該變量在事件組結構體中定義，事件標誌組的可用位數通過宏定義確定：

  • configUSE_16_BIT_TICKS定義為1，uxEventBits是16位的，低8位用來存儲事件組
  • configUSE_16_BIT_TICKS 定義為0，uxEventBits是32位的，低24位用來存儲事件組

註：高8位用於系統內核使用，不可用戶使用！

#if configUSE_16_BIT_TICKS == 1
    #define eventCLEAR_EVENTS_ON_EXIT_BIT   0x0100U
    #define eventUNBLOCKED_DUE_TO_BIT_SET   0x0200U
    #define eventWAIT_FOR_ALL_BITS          0x0400U
    #define eventEVENT_BITS_CONTROL_BYTES   0xff00U
#else
    #define eventCLEAR_EVENTS_ON_EXIT_BIT   0x01000000UL     /*在退出時清除位*/
    #define eventUNBLOCKED_DUE_TO_BIT_SET   0x02000000UL
    #define eventWAIT_FOR_ALL_BITS          0x04000000UL     /*等待所有位*/
    #define eventEVENT_BITS_CONTROL_BYTES   0xff000000UL
#endif

先放一張整體結構圖：

再來看源碼：

EventBits_t xEventGroupWaitBits( EventGroupHandle_t xEventGroup, const EventBits_t uxBitsToWaitFor, const BaseType_t xClearOnExit, const BaseType_t xWaitForAllBits, TickType_t xTicksToWait )
{
    EventGroup_t *pxEventBits = ( EventGroup_t * ) xEventGroup;
    EventBits_t uxReturn, uxControlBits = 0;
    BaseType_t xWaitConditionMet, xAlreadyYielded;
    BaseType_t xTimeoutOccurred = pdFALSE;

    configASSERT( xEventGroup );
    /* 斷言，判斷要設置的事件標誌位是否有效，防止用戶使用高8位 */
    configASSERT( ( uxBitsToWaitFor & eventEVENT_BITS_CONTROL_BYTES ) == 0 );
    configASSERT( uxBitsToWaitFor != 0 );
    #if ( ( INCLUDE_xTaskGetSchedulerState == 1 ) || ( configUSE_TIMERS == 1 ) )
    {
        configASSERT( !( ( xTaskGetSchedulerState() == taskSCHEDULER_SUSPENDED ) && ( xTicksToWait != 0 ) ) );
    }
    #endif

    /*先停止任務調度*/
    vTaskSuspendAll();
    {
        const EventBits_t uxCurrentEventBits = pxEventBits->uxEventBits;

        /* 先看下當前事件中的標誌位是否已經滿足條件了 */
        xWaitConditionMet = prvTestWaitCondition( uxCurrentEventBits, uxBitsToWaitFor, xWaitForAllBits );

        /* 滿足條件了*/
        if( xWaitConditionMet != pdFALSE )
        {
            /*直接返回，注意這裡返回的是的當前事件的[所有]標誌位 */
            uxReturn = uxCurrentEventBits;
            /*等待時間強制置0*/
            xTicksToWait = ( TickType_t ) 0;

            /* 若設置了退出的時候需要清除對應的事件標誌位 */
            if( xClearOnExit != pdFALSE )
            {
                /*清除對應的標誌位*/
                pxEventBits->uxEventBits &= ~uxBitsToWaitFor;
            }
            else
            {
                mtCOVERAGE_TEST_MARKER();
            }
        }
        /* 不滿足條件，且設置的是不等待*/
        else if( xTicksToWait == ( TickType_t ) 0 )
        {
            /*也是返回當前事件的[所有]標誌位 */
            uxReturn = uxCurrentEventBits;
        }
        /* 不滿足條件，且用戶指定了超時時間*/
        else
        {
            /* 若設置了退出的時候需要清除對應的事件標誌位 */
            if( xClearOnExit != pdFALSE )
            {   
                /* 保存一下當前任務的信息標記，以便在恢復任務的時候對事件進行相應的操作 */
                uxControlBits |= eventCLEAR_EVENTS_ON_EXIT_BIT; /*0x01000000UL 退出時清除位*/
            }
            else
            {
                mtCOVERAGE_TEST_MARKER();
            }

            /*若需要等待所有事件*/
            if( xWaitForAllBits != pdFALSE )
            {
                /* 保存一下當前任務的信息標記，以便在恢復任務的時候對事件進行相應的操作 */
                uxControlBits |= eventWAIT_FOR_ALL_BITS; /*0x04000000UL 等待所有位*/
            }
            else
            {
                mtCOVERAGE_TEST_MARKER();
            }

            /* 當前任務進入該事件組的「事件等待列表」中，任務將被阻塞指定時間xTicksToWait !!! */
            vTaskPlaceOnUnorderedEventList( &( pxEventBits->xTasksWaitingForBits ), ( uxBitsToWaitFor | uxControlBits ), xTicksToWait );

            uxReturn = 0;

            traceEVENT_GROUP_WAIT_BITS_BLOCK( xEventGroup, uxBitsToWaitFor );
        }
    }
     /*恢復任務調度*/
    xAlreadyYielded = xTaskResumeAll();

    /*xTicksToWait為0時不執行(1.本身設置的為0  2.符合的事件觸發後被清0)*/
    if( xTicksToWait != ( TickType_t ) 0 )
    {
        /*恢復任務後還沒有進行任務切換*/
        if( xAlreadyYielded == pdFALSE )
        {
            /* 進行一次任務切換!!! */
            portYIELD_WITHIN_API();
        }
        else
        {
            mtCOVERAGE_TEST_MARKER();
        }

        /* 進入到這裡說明當前的任務已經被重新調度了!!!（1.符合的事件被觸發  2.事件等待超時） */
        uxReturn = uxTaskResetEventItemValue();

        if( ( uxReturn & eventUNBLOCKED_DUE_TO_BIT_SET ) == ( EventBits_t ) 0 )
        {
            /*進入臨界區*/
            taskENTER_CRITICAL();
            {
                /* 超時返回時，直接返回當前事件的[所有]標誌位!!! */
                uxReturn = pxEventBits->uxEventBits;

                /* 再判斷一次是否發生了事件 */
                if( prvTestWaitCondition( uxReturn, uxBitsToWaitFor, xWaitForAllBits ) != pdFALSE )
                {
                    /* 若設置了退出的時候需要清除對應的事件標誌位 */
                    if( xClearOnExit != pdFALSE )
                    {   
                        /* 清除事件標誌位並且返回 */
                        pxEventBits->uxEventBits &= ~uxBitsToWaitFor;
                    }
                    else
                    {
                        mtCOVERAGE_TEST_MARKER();
                    }
                }
                else
                {
                    mtCOVERAGE_TEST_MARKER();
                }
            }
            /*退出臨界區*/
            taskEXIT_CRITICAL();

            /* 在未使用跟蹤宏時防止編譯器警告 */
            xTimeoutOccurred = pdFALSE;
        }
        else
        {
            /* 因為已經設置了位，所以任務解除了阻塞 */
        }

        /* 清除內核使用的事件位（高8位 ） */
        uxReturn &= ~eventEVENT_BITS_CONTROL_BYTES;/*0xff000000UL*/
    }
    traceEVENT_GROUP_WAIT_BITS_END( xEventGroup, uxBitsToWaitFor, xTimeoutOccurred );

    return uxReturn;
}

注意這裡的參數：

  • xEventGroup：事件標誌組句柄
  • uxBitsToWaitFor：等待 24 個事件標誌位中的指定標誌
  • xClearOnExit：是否清除已經被置位的事件標誌
  • 如果這個參數設置為 pdTRUE，且函數xEventGroupWaitBits 在參數 xTicksToWait 設置的溢出時間內返回，那麼相應被設置的事件標誌位會被清零。
  • 如果這個參數設置為 pdFALSE，對已經被設置的事件標誌位沒有影響。
  • xWaitForAllBits：是否等待所有的標誌位都被設置
  • 如果這個參數設置為 pdTRUE，要等待第 2 個參數 uxBitsToWaitFor 所指定的標誌位全部被置 1，函數才可以返回。當然，超出了在參數xTicksToWait 設置的溢出時間也是會返回的。
  • 如果這個參數設置為 pdFALSE，第 2 個參數uxBitsToWaitFor 所指定的任何標誌位被置 1，函數都會返回，超出溢出時間也會返回。
  • xTicksToWait ：設置等待時間，單位時鐘節拍週期。 如果設置為 portMAX_DELAY，表示永久等待。

對於返回值：

  • 如果由於設置的等待時間超時，函數的返回值可能會有部分事件標誌位被置 1
  • 如果由於指定的事件標誌位被置1而返回， 並且設置了這個函數的參數xClearOnExit為pdTRUE，那麼此函數的返回值是清零前的事件標誌組數值
  • 調用此函數的任務在離開阻塞狀態到退出函數 xEventGroupWaitBits 之間這段時間，如果一個高優先級的任務搶佔執行了，並且修改了事件標誌位，那麼此函數的返回值會跟當前的事件標誌組數值不同

2.3.1 檢查等待的條件

等待事件中，要檢查等待的條件是否滿足要求

static BaseType_t prvTestWaitCondition( const EventBits_t uxCurrentEventBits, const EventBits_t uxBitsToWaitFor, const BaseType_t xWaitForAllBits )
{
    BaseType_t xWaitConditionMet = pdFALSE;

    /*不是等待所有事件*/
    if( xWaitForAllBits == pdFALSE )
    {
        /* 當前事件狀態存在某一等待事件 */
        if( ( uxCurrentEventBits & uxBitsToWaitFor ) != ( EventBits_t ) 0 )
        {   
            /*返回結果設為TRUE*/
            xWaitConditionMet = pdTRUE;
        }
        else
        {
            mtCOVERAGE_TEST_MARKER();
        }
    }
    else/*等待所有事件*/
    {
        /* 當前事件狀態等於要等待的所有事件 */
        if( ( uxCurrentEventBits & uxBitsToWaitFor ) == uxBitsToWaitFor )
        {   
            /*返回結果設為TRUE*/
            xWaitConditionMet = pdTRUE;
        }
        else
        {
            mtCOVERAGE_TEST_MARKER();
        }
    }

    return xWaitConditionMet;
}

2.3.2 將等待任務加入等待列表（無序列表）

等待事件中，若此次事件還不滿足等待條件，則先將任務加入等待列表繼續等待

/*將等待任務加入事件組的「事件等待列表」  參數：等待列表，信息參數，等待時間*/
void vTaskPlaceOnUnorderedEventList( List_t * pxEventList, const TickType_t xItemValue, const TickType_t xTicksToWait )
{
    configASSERT( pxEventList );

    /* 此函數必須在掛起調度程序的情況下調用，它由事件組實現使用 */
    configASSERT( uxSchedulerSuspended != 0 );

    /* 存儲Item值在事件列表項中。在這裡訪問事件列表項是安全的，因為中斷不會訪問未處於阻塞狀態的任務的事件列表項 */
    listSET_LIST_ITEM_VALUE( &( pxCurrentTCB->xEventListItem ), xItemValue | taskEVENT_LIST_ITEM_VALUE_IN_USE ); /*0x80000000UL*/

    /* 將TCB的事件列表項放在事件列表的末尾。在這裡訪問事件列表是安全的，因為它是事件組的一部分——中斷不直接訪問事件組(相反，它們通過掛起任務級別的函數調用來間接訪問事件組) */
    vListInsertEnd( pxEventList, &( pxCurrentTCB->xEventListItem ) );

    /* 添加任務到延時列表 */
    prvAddCurrentTaskToDelayedList( xTicksToWait, pdTRUE );
}

    if( uxListRemove( &( pxCurrentTCB->xStateListItem ) ) == ( UBaseType_t ) 0 )

    }
    else
    {
        mtCOVERAGE_TEST_MARKER();
    }

    #if ( INCLUDE_vTaskSuspend == 1 )

        if( ( xTicksToWait == portMAX_DELAY ) && ( xCanBlockIndefinitely != pdFALSE ) )

            /* 將任務添加到掛起的任務列表而不是延遲的任務列表中，以確保它不會被計時事件喚醒。它會無限期地阻塞 */

                vListInsert( pxOverflowDelayedTaskList, &( pxCurrentTCB->xStateListItem ) );
            }

                vListInsert( pxDelayedTaskList, &( pxCurrentTCB->xStateListItem ) );

                /* 如果進入阻塞狀態的任務位於阻塞任務列表的最前面，那麼xNextTaskUnblockTime也需要更新 */

2.4 設置事件組標誌位(非中斷中)

先放一張整體結構圖：

再來看源碼：

EventBits_t xEventGroupSetBits( EventGroupHandle_t xEventGroup, const EventBits_t uxBitsToSet )
{
    ListItem_t *pxListItem, *pxNext;
    ListItem_t const *pxListEnd;
    List_t *pxList;
    EventBits_t uxBitsToClear = 0, uxBitsWaitedFor, uxControlBits;
    EventGroup_t *pxEventBits = ( EventGroup_t * ) xEventGroup; /*指定的事件標誌組*/
    BaseType_t xMatchFound = pdFALSE;

    configASSERT( xEventGroup );
    /* 斷言，判斷要設置的事件標誌位是否有效，防止用戶使用高8位 */
    configASSERT( ( uxBitsToSet & eventEVENT_BITS_CONTROL_BYTES ) == 0 );/*0xff000000UL*/

    /*獲取事件標誌組的等待位列表*/
    pxList = &( pxEventBits->xTasksWaitingForBits ); 

    /*獲取列表的末尾項*/
    pxListEnd = listGET_END_MARKER( pxList ); 

    /*先停止任務調度*/
    vTaskSuspendAll();
    {
        traceEVENT_GROUP_SET_BITS( xEventGroup, uxBitsToSet );

        /*獲取列表的首項*/
        pxListItem = listGET_HEAD_ENTRY( pxList ); 

        /* 指定的事件標誌組的事件位，設置事件標誌 */
        pxEventBits->uxEventBits |= uxBitsToSet;

        /* 設置這個事件標誌位可能是某個任務在等待的事件，就遍歷等待事件列表中的任務 */
        while( pxListItem != pxListEnd )
        {
            /*從列表首項開始檢查*/
            pxNext = listGET_NEXT( pxListItem );
            /*獲取列表的值*/
            uxBitsWaitedFor = listGET_LIST_ITEM_VALUE( pxListItem );
            xMatchFound = pdFALSE;

            /* 將等待的位從控制位中分離出來*/  /*0xff000000UL*/
            uxControlBits = uxBitsWaitedFor & eventEVENT_BITS_CONTROL_BYTES;  /*控制位*/
            uxBitsWaitedFor &= ~eventEVENT_BITS_CONTROL_BYTES;                /*等待位*/

            /* 如果只需要有一個事件標誌位滿足即可(等待所有的位沒有被標記) */ /*0x04000000UL*/
            if( ( uxControlBits & eventWAIT_FOR_ALL_BITS ) == ( EventBits_t ) 0 )
            {
                /*判斷要等待的事件是否發生了*/
                if( ( uxBitsWaitedFor & pxEventBits->uxEventBits ) != ( EventBits_t ) 0 )
                {
                    xMatchFound = pdTRUE; /*事件符合*/
                }
                else
                {
                    mtCOVERAGE_TEST_MARKER();
                }
            }
            /*否則就要所有事件都發生的時候才能解除阻塞 */
            else if( ( uxBitsWaitedFor & pxEventBits->uxEventBits ) == uxBitsWaitedFor )
            {
                /* 所有事件都發生了 */
                xMatchFound = pdTRUE; /*事件符合*/
            }
            else
            {
                /* Need all bits to be set, but not all the bits were set. */
            }

            if( xMatchFound != pdFALSE )
            {
                /* 匹配了標誌位，然後看下是否需要清除標誌位*/
                if( ( uxControlBits & eventCLEAR_EVENTS_ON_EXIT_BIT ) != ( EventBits_t ) 0 )
                {
                    /*記錄下需要清除的標誌位，等遍歷完隊列之後統一處理*/
                    uxBitsToClear |= uxBitsWaitedFor;
                }
                else
                {
                    mtCOVERAGE_TEST_MARKER();
                }

                /* 將滿足事件條件的任務從等待列表中移除，並且添加到就緒列表中 */
                ( void ) xTaskRemoveFromUnorderedEventList( pxListItem, pxEventBits->uxEventBits | eventUNBLOCKED_DUE_TO_BIT_SET );
            }

            /* 循環遍歷事件等待列表，可能不止一個任務在等待這個事件!!! */
            pxListItem = pxNext;
        }

        /* 遍歷完畢，清除事件標誌位 */ 
        pxEventBits->uxEventBits &= ~uxBitsToClear;
    }
    ( void ) xTaskResumeAll();

    return pxEventBits->uxEventBits;
}

注意，此函數的參數uxBitsToSet是要設置的事件位，函數返回一個EventBits_t的數據，它是當前事件組中事件位的值，但uxBitsToSet與EventBits_t的值可能不相同:

  • 如果設置一個事件位後，使得等待該位的任務離開阻塞狀態（注意是離開阻塞態，即使沒有進入運行態，只要離開阻塞態即可），則該位可能會被自動清除
  • 調用此函數的任務是一個低優先級任務，通過此函數設置了事件標誌後，讓一個等待此事件標誌的高優先級任務就緒了，會立即切換到高優先級任務去執行，相應的事件標誌位會被函數xEventGroupWaitBits清除掉，等從高優先級任務返回到低優先級任務後，函數xEventGroupSetBits 的返回值已經被修改

2.4.1 將滿足事件條件的任務從等待列表中移除

BaseType_t xTaskRemoveFromUnorderedEventList( ListItem_t * pxEventListItem, const TickType_t xItemValue )
{
TCB_t *pxUnblockedTCB;
BaseType_t xReturn;

    /*如果進入阻塞狀態的任務位於阻塞任務列表的最前面，那麼xNextTaskUnblockTime也需要更新。
    必須在掛起調度程序的情況下調用此函數。它由事件標誌實現使用。 */
    configASSERT( uxSchedulerSuspended != pdFALSE );

    /* 將新項值存儲在事件列表中 */
    listSET_LIST_ITEM_VALUE( pxEventListItem, xItemValue | taskEVENT_LIST_ITEM_VALUE_IN_USE );

    /* 從事件標誌中刪除事件列表。中斷不訪問事件標誌 */
    pxUnblockedTCB = ( TCB_t * ) listGET_LIST_ITEM_OWNER( pxEventListItem );
    configASSERT( pxUnblockedTCB );
    ( void ) uxListRemove( pxEventListItem );

    /* 從延遲列表中刪除任務並將其添加到就緒列表中。調度器被掛起，因此中斷將不會訪問就緒列表 */
    ( void ) uxListRemove( &( pxUnblockedTCB->xStateListItem ) );
    prvAddTaskToReadyList( pxUnblockedTCB );

    if( pxUnblockedTCB->uxPriority > pxCurrentTCB->uxPriority )
    {
        /* 如果從事件列表中刪除的任務的優先級高於調用任務，則返回true。
        這允許調用任務知道它現在是否應該強制上下文切換。 */
        xReturn = pdTRUE;

        /* 在用戶沒有使用ISR安全FreeRTOS函數的「xHigherPriorityTaskWoken」參數的情況下，
        將yield標記為pending */
        xYieldPending = pdTRUE;
    }
    else
    {
        xReturn = pdFALSE;
    }

    return xReturn;
}

2.5 設置事件組標誌位(中斷中)

#if ( ( configUSE_TRACE_FACILITY == 1 ) && ( INCLUDE_xTimerPendFunctionCall == 1 ) && ( configUSE_TIMERS == 1 ) )

    BaseType_t xEventGroupSetBitsFromISR( EventGroupHandle_t xEventGroup, const EventBits_t uxBitsToSet, BaseType_t *pxHigherPriorityTaskWoken )
    {
    BaseType_t xReturn;

        traceEVENT_GROUP_SET_BITS_FROM_ISR( xEventGroup, uxBitsToSet );
        xReturn = xTimerPendFunctionCallFromISR( vEventGroupSetBitsCallback, ( void * ) xEventGroup, ( uint32_t ) uxBitsToSet, pxHigherPriorityTaskWoken );

        return xReturn;
    }

#endif

由於該函數對事件標誌組的操作是不確定性操作，因為不知道當前有多少個任務在等待此事件標誌。而FreeRTOS 不允許在中斷服務程序和臨界段中執行不確定性操作。

為了不在中斷服務程序中執行，就通過此函數給 FreeRTOS 的 daemon 任務（即軟件定時器任務）發送消息，在 daemon 任務中執行事件標誌的置位操作。 同時也為了不在臨界段中執行此不確定操作，將臨界段改成由調度鎖來完成。這樣不確定性操作在中斷服務程序和臨界段中執行的問題就都得到解決了。

關於daemon 任務（軟件定時器任務）的介紹，可查看之前的文章：

    }

再來看一下回調函數，其是就是非中斷方式的事件標誌組置位，所以，中斷方式的事件標誌組置位，是通過消息隊列告知Daemon任務，然後在Daemon任務中實現置維操作。

3 使用示例

3.1 測試1

//事件組觸發任務（按鍵）
#define EVENTSETBIT_TASK_PRIO   2
#define EVENTSETBIT_STK_SIZE    256  
TaskHandle_t EventSetBit_Handler;
void eventsetbit_task(void *pvParameters);

//事件組等待任務
#define EVENTGROUP_TASK_PRIO    3
#define EVENTGROUP_STK_SIZE     256  
TaskHandle_t EventGroupTask_Handler;
void eventgroup_task(void *pvParameters);

#define KEY1_EVENT    (1<<1)    //按鍵KEY1置位bit1
#define KEY0_EVENT    (1<<0)    //按鍵KEY0置位bit0
#define KEY_UP_EVENT  (1<<2)    //按鍵KEY_UP置位bit2
#define EVENT_1_0   (KEY1_EVENT|KEY0_EVENT) //等待KEY1與KEY0

//設置事件位的任務
void eventsetbit_task(void *pvParameters)
{
    u8 key;
    EventBits_t uxBits;
    while(1)
    {
        if(EventGroupHandler!=NULL)
        {
            key=KEY_Scan(0);
            switch(key)
            {
                case KEY1_PRES:
                    uxBits = xEventGroupSetBits(EventGroupHandler,KEY1_EVENT);
                    if((uxBits & KEY1_EVENT) != 0)
                    {
                        printf("K1鍵按下，事件標誌的bit1被設置\r\n");
                    }
                    else
                    {
                        printf("K1鍵按下，事件標誌的bit1被清除，說明任務已經接收到bit0和bit1被設置的情況\r\n");
                    }
                    break;
                case KEY0_PRES:
                    uxBits = xEventGroupSetBits(EventGroupHandler,KEY0_EVENT);
                    if((uxBits & KEY0_EVENT) != 0)
                    {
                        printf("K0鍵按下，事件標誌的bit0被設置\r\n");
                    }
                    else
                    {
                        printf("K0鍵按下，事件標誌的bit0被清除，說明任務已經接收到bit0和bit1被設置的情況\r\n");
                    }
                    break;
                case WKUP_PRES:
                    printf("K_UP鍵按下，事件標誌的bit2被設置，這時一個無用事件\r\n");
                    xEventGroupSetBits(EventGroupHandler,KEY_UP_EVENT);
                    break;  
            }
        }
        vTaskDelay(10); //延時10ms，也就是10個時鐘節拍
    }
}

//等待事件標誌組並執行處理的任務
void eventgroup_task(void *pvParameters)
{
    EventBits_t uxBits;
    while(1)
    {

        if(EventGroupHandler!=NULL)
        {
            //等待事件組中的相應事件位
            uxBits=xEventGroupWaitBits((EventGroupHandle_t  )EventGroupHandler,     
                                           (EventBits_t         )EVENT_1_0, //等待bit1與bit0
                                           (BaseType_t          )pdTRUE, //退出前清除事件標誌                
                                           (BaseType_t          )pdTRUE, //等待bit1與bit0都被置位
                                           (TickType_t          )5000);  //最多阻塞5秒
            if((uxBits & EVENT_1_0) == EVENT_1_0)
            {
                printf("等待ok,接收到bit0和bit1：事件標誌組的值:%d\r\n",uxBits);
            }
            else
            {
                printf("等待超時：事件標誌組的值:%d\r\n",uxBits);
            }
            LED1=!LED1; 
        }
        else
        {
            vTaskDelay(10);
        }
    }
}

注意這裡的一些參數，等待事件是要等待KEY1與KEY0都被置位，成功等待到事件後會把自動清除事件標誌，每次等待的事件最大為5秒。

另外，這裡設置等待事件的任務要比按鍵觸發事件任務的優先級高。

測試結果如下：

等待超時：事件標誌組的值:0
K1鍵按下，事件標誌的bit1被設置
等待ok,接收到bit0和bit1：事件標誌組的值:3
K0鍵按下，事件標誌的bit0被清除，說明任務已經接收到bit0和bit1被設置的情況

K_UP鍵按下，事件標誌的bit2被設置，這時一個無用事件
等待超時：事件標誌組的值:4
K0鍵按下，事件標誌的bit0被設置
等待ok,接收到bit0和bit1：事件標誌組的值:7
K1鍵按下，事件標誌的bit1被清除，說明任務已經接收到bit0和bit1被設置的情況
等待超時：事件標誌組的值:4

先看第1段的結果

  • 首先，如何一直沒有任何按鍵按下，則5秒後，等待超時，則會打印出等待超時，且當前的事件標誌組的值就是0
  • 然後，KEY1按下，觸發事件位bit1，此時xEventGroupWaitBits會得到bit1，由於需要等待bit1與bit0都被置位，所以仍會繼續等待
  • 接著，當KEY0按下時，觸發事件位bit0，此時bit1與bit0都被置位，由於等待任務的優先級高，所以在條件滿足後，會切換到等待任務，打印出等待ok，並清除對應的事件事件標誌，打印出的「事件標誌組的值:3」而不是清除後的0，是因為返回值的清除前的值
  • 最後，調度器再次回到低優先級的按鍵處理任務，因為此時標誌位已經被清除了，所以打印出「K0鍵按下，事件標誌的bit0被清除」

再看第2段的結果

  • 首先，按下K_UP鍵，這個事件不是我們想等待的，在5秒後，等待超時，但依然會返回這個不需要的等待值
  • 然後，依次按下KEY0和KEY1，這時xEventGroupWaitBits等到了bit0、bit1與bit2，滿足等待條件，但清除標誌時只會清除自己等待的bit0和bit1
  • 所以，再過5秒後，雖然沒有再次按下KEY_UP鍵，但bit2的值仍然保留在之間標誌組中

先來看第1段結果

  • 首先，KEY1按下，打印第1行
  • 然後KEY0按下，由於將按鍵任務的優先級調高，此時調度器不會立即切換到低優先級的事件等待任務，而會繼續執行自身直到被阻塞，所以會有第2行的打印，但是注意，第2行按下KEY0的打印卻依舊顯示的是被清除了，因為在等待任務中使用了事件等待，而KEY0按下的時候，FreeRTOS操作系統會知道等待兩個按鍵按下的事件已經觸發了，此時，在按鍵任務中，xEventGroupSetBits的返回值，也不是當前獲取的置位值了，而是經過xEventGroupSetBits函數自動清零之後的值，所以第2行打印的是清零信息
  • 最後，第3行打印的是都被置位，為什麼不是清零？因為此時的xEventGroupWaitBits返回值是清零前的事件標誌組的值

再看第2段的結果

  • 首先，第1行的等待超時時，事件組的數值是0
  • 然後，只按下KEY1，讓等待超時
  • 最後，當等待超時後，雖然我們設置的退出時清除事件標誌，但因為沒有同時等待到bit1與bit0，所以超時退出後不會清除單一的事件標誌

  • KEY_0按下，同理
  • KEY_UP按下，不是想要等待的事件，會執行到超時等待，並且bit2始終沒有被清除（如果需要清除可通過xEventGroupClearBits手動清除）

4 總結與注意事項

  • 事件標誌組可以指定要等待哪些事件，但如果是其它事件觸發了，在等待超時後，這些不希望等待的事件也會在超時後作為等待的返回結果
  • 由於函數 xEventGroupSetBitsFromISR 對事件標誌的置位操作是在 daemon 任務裡面執行的，如果想讓置位操作立即生效，即讓等此事件標誌的任務能夠得到及時執行，需要設置 daemon 任務的優先級高於使用此事件標誌組的所有其它任務
  • 設置自動清除，只能清除要等待的位，如果是等到多個事件同時發送，當只有部分事件發送且等到超時後，事件位是不會被自動清除的，只能等之後所有事件都觸發後才能自動清除，或通過手工清除
